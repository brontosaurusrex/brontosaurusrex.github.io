## A script to populate folders by size

    #!/bin/bash

    # populate

    cd="1"
    mkdir -p "$cd"

    # main
    cat byDate.txt | while read -r line; do 

        echo "$cd $line"
        
        size="$(du -m "$cd" | cut -f1)"
        
        if (( size > 690 )); then
        
            rm "$cd/$prev"
        
            ((cd=cd+1))
            mkdir -p "$cd"
            
            cp "$prev" "$cd"
            
        else
            
            cp "$line" "$cd"
            
        fi
        
        # store previous line
        prev="$line"
        
        #sleep 1
        
    done
